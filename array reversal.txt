#include <stdio.h>
#include <math.h>

int main()
{
    int a[1000],n,i;
    scanf("%d",&a[i]);
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    for(i=n;i>0;i--)
    {
        printf("%d",a[i]);
    }
}
